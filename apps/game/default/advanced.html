Deployment Configuration

apiVersion: extensions/v1beta1
kind: Deployment
metadata:
    name: api-deployment
spec:
    replicas: 12
    strategy:
        type: RollingUpdate
    template:
        metadata:
            labels:
                app: api
        spec:
            containers:
            - name: "api"
              image: "gcr.io/projectid/api"
              ports:
              - name: "http"
                containerPort: 8080
                protocol: TCP

Service Configuration

apiVersion: v1
kind: Service
metadata:
    labels:
        name: api
    name: api
spec:
    type: LoadBalancer
    loadBalancerIP: 35.198.214.114
    ports:
    - name: "http"
        port: 80
        targetPort: 8080
        protocol: TCP
    selector:
        app: api
